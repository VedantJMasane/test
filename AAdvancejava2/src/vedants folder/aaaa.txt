@Service
public class UserService {
    @Autowired
    private UserRepository userRepository;

    @Transactional
    public User createUserWithCustomer(User user, Customer customer) {
        user.setCustomer(customer);
        customer.setUser(user); // Bidirectional link
        return userRepository.save(user);
    }
